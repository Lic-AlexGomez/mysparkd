package sparkd.DTOs;

import java.time.LocalDateTime;
import java.util.UUID;
import sparkd.Enums.MediaType;

public class MessageResponseDTO {

    private UUID id;
    private UUID chatId;
    private UUID senderId;
    private UUID receiverId;
    private String content;
    private LocalDateTime sentAt;
    private MediaType mediaType;
    private MediaDTO media;
    private boolean read;   // <-- NUEVO

    public MessageResponseDTO(
            UUID id,
            UUID chatId,
            UUID senderId,
            UUID receiverId,
            String content,
            LocalDateTime sentAt,
            MediaType mediaType,
            boolean read        // <-- NUEVO
    ) {
        this.id = id;
        this.chatId = chatId;
        this.senderId = senderId;
        this.receiverId = receiverId;
        this.content = content;
        this.sentAt = sentAt;
        this.mediaType = mediaType;
        this.read = read;       // <-- NUEVO
    }

    public UUID getId() { return id; }
    public UUID getChatId() { return chatId; }
    public UUID getSenderId() { return senderId; }
    public UUID getReceiverId() { return receiverId; }
    public String getContent() { return content; }
    public LocalDateTime getSentAt() { return sentAt; }
    public MediaType getMediaType() { return mediaType; }
    public MediaDTO getMedia() { return media; }
    public void setMedia(MediaDTO media) { this.media = media; }
    public boolean isRead() { return read; }   // <-- NUEVO
}
